## 默认页面

```vue
<script lang="ts">
import { buildSortingField } from '@/utils'
import { dateFormatter } from '@/utils/formatTime'
import { useDefaultStationQueryStore } from '@/store/modules/defaultStationQuery'
import { usePageTableHeight } from '@/hooks/web/useTableHeight'
import { useVisibleSearchForm } from '@/components/Search'
import * as Api from '../services/api'
import * as Types from '../services/types'
</script>

<script lang="ts" setup>
defineOptions({ name: 'XxxXxx' })

/** 消息弹窗 */
const message = useMessage()
/** 国际化 */
const { t } = useI18n()

/** 搜索表单可见性 */
const { isVisibleSearch, VisibleSearchButton } = useVisibleSearchForm()
/** 列表加载状态 */
const loading = ref(true)
/** 列表的总页数 */
const total = ref(0)
/** 列表的数据 */
const list = ref<Types.Xxx[]>([])
/** 搜索的表单 */
const queryFormRef = ref()
/** 列表查询条件 */
const queryParams = reactive({
  pageNo: 1,
  pageSize: 20,
  createTime: [],
  sortingFields: [] as { field: string; order: string }[]
})
/** 表格高度 */
const { tableHeight } = usePageTableHeight(queryFormRef)
/**
 * 查询列表
 * @author AhJindeg <AhJindeg@gmail.com>
 **/
async function getList() {
  loading.value = true
  const data = await Api.getXxxPage(queryParams).catch(() => ({
    list: [],
    total: 0
  }))
  list.value = data.list
  total.value = data.total
  loading.value = false
}

/**
 * 搜索按钮操作
 * @author AhJindeg <AhJindeg@gmail.com>
 **/
function handleQuery() {
  queryParams.pageNo = 1
  getList()
}

/**
 * 重置按钮操作
 * @author AhJindeg <AhJindeg@gmail.com>
 **/
function resetQuery() {
  queryFormRef.value.resetFields()
  handleQuery()
}

/**
 * 排序
 * @author AhJindeg <AhJindeg@gmail.com>
 **/
function handleSortChange(params: { prop: string; order: string | null }) {
  queryParams.sortingFields = [buildSortingField(params)]
  handleQuery()
}

/**
 * 删除按钮操作
 * @author AhJindeg <AhJindeg@gmail.com>
 **/
async function handleDelete(id: number) {
  try {
    // 删除的二次确认
    await message.delConfirm()
    // 发起删除
    await Api.deleteXxx(id)
    message.success(t('common.delSuccess'))
  } finally {
    await getList()
  }
}

/** 导出的加载中 */
const exportLoading = ref(false)
/**
 * 导出按钮操作
 * @author AhJindeg <AhJindeg@gmail.com>
 **/
async function handleExport() {
  try {
    // 导出的二次确认
    await message.exportConfirm()
    // 发起导出
    exportLoading.value = true
    const data = await Api.exportXxx(queryParams)
    download.excel(data, `${t('x.x.menuName')}.xls`)
  } finally {
    exportLoading.value = false
  }
}

onMounted(() => {
  /** 全局默认筛查站点store */
  const defaultStationQueryStore = useDefaultStationQueryStore()
  queryParams.stationId = defaultStationQueryStore.getStationId
  debouncedStationList(queryParams.stationId)
  getList()
})
</script>

<template>
  <ContentWrap>
    <!-- 搜索工作栏 -->
    <el-form class="zykw-query-form zykw-query-form-gap-30" :model="queryParams" ref="queryFormRef">
      <el-form-item class="col-span-6" :label="t('x.x.x')" prop="x">
        <el-input
          class="!flex-1"
          v-model="queryParams.x"
          :placeholder="t('x.x.x')"
          clearable
          @keyup.enter="handleQuery"
        />
      </el-form-item>
      <el-form-item
        v-show="isVisibleSearch"
        class="col-span-6"
        :label="t('x.x.x')"
        prop="x"
      >
        <el-select
          class="!flex-1"
          v-model="queryParams.x"
          :placeholder="t('x.x.x')"
          clearable
        >
          <el-option
            v-for="{ value, label } in getStrDictOptions(DICT_TYPE.XXX)"
            :key="value"
            :label="label"
            :value="value"
          />
        </el-select>
      </el-form-item>
      <el-form-item
        v-show="isVisibleSearch"
        class="col-span-6"
        :label="t('common.editTime')"
        prop="updateTime"
      >
        <el-date-picker
          class="!flex-1"
          v-model="queryParams.updateTime"
          value-format="YYYY-MM-DD HH:mm:ss"
          type="daterange"
          :start-placeholder="t('common.startDate')"
          :end-placeholder="t('common.endDate')"
          :default-time="[new Date('1 00:00:00'), new Date('1 23:59:59')]"
        />
      </el-form-item>
      <el-form-item class="col-span-12">
        <el-button type="primary" plain @click="handleQuery"
          ><Icon icon="ep:search" class="mr-5px" />{{ t('common.search') }}</el-button
        >
        <el-button @click="resetQuery"
          ><Icon icon="ep:refresh" class="mr-5px" />{{ t('common.reset') }}</el-button
        >
        <el-button
          type="primary"
          @click="openForm('create')"
          v-hasPermi="['x:x:x']"
        >
          <Icon icon="ep:plus" class="mr-5px" /> {{ t('common.create') }}
        </el-button>
        <el-button
          type="success"
          plain
          @click="handleExport"
          :loading="exportLoading"
          v-hasPermi="['x:x:x']"
        >
          <Icon icon="ep:download" class="mr-5px" /> {{ t('common.export') }}
        </el-button>
        <VisibleSearchButton />
      </el-form-item>
    </el-form>
    <el-table class="zykw-table" v-loading="loading" :data="list" border :height="tableHeight" @sort-change="handleSortChange">
      <el-table-column :label="t('common.index')" align="center" type="index" width="60" />
      <el-table-column
        :label="t('x.x.x')"
        prop="x"
        min-width="160"
        show-overflow-tooltip
      />
      <el-table-column
        :label="t('x.x.x')"
        prop="x"
        min-width="100"
        show-overflow-tooltip
      >
        <template #default="{ row }">
          {{ getDictLabel(DICT_TYPE.XXX, row.x) }}
        </template>
      </el-table-column>
      <el-table-column
        :label="t('x.x.x')"
        prop="x"
        width="120"
        show-overflow-tooltip
      >
        <template #default="{ row }">
          <dict-tag :type="DICT_TYPE.XXX" :value="row.x" />
        </template>
      </el-table-column>
      <el-table-column
        :label="t('common.editTime')"
        prop="updateTime"
        width="180"
        :formatter="dateFormatter"
        sortable="custom"
      />
      <el-table-column :label="t('common.operation')" width="270" align="center" fixed="right">
        <template #default="{ row }">
          <el-button
            link
            type="primary"
            @click="openForm('detail', row.id)"
            v-hasPermi="['x:x:x']"
          >
            {{ t('common.detail') }}
          </el-button>
          <el-button
            link
            type="primary"
            @click="openForm('update', row.id)"
            v-hasPermi="['x:x:x']"
          >
            {{ t('common.update') }}
          </el-button>
          <el-button
            link
            type="danger"
            @click="handleDelete(row.id)"
            v-hasPermi="['x:x:x']"
          >
            {{ t('common.delete') }}
          </el-button>
        </template>
      </el-table-column>
    </el-table>
    <!-- 分页 -->
    <Pagination
      :total="total"
      v-model:page="queryParams.pageNo"
      v-model:limit="queryParams.pageSize"
      @pagination="getList"
    />
  </ContentWrap>
</template>
```
